export async function registerTeam(data: any) {

    body: JSON.stringify(data),
  }).then(res => res.json());
}

export async function getCertificate(email: string) {
  return fetch(API_BASE + "/certificate?email=" + encodeURIComponent(email))
    .then(res => res.json());
